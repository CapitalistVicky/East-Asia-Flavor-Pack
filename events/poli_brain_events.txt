namespace = poli_brain_events

#반년마다 정통성 체크

poli_brain_events.1 = {
	type = country_event

	placement = root
	
	#duration = 1
	#is_popup = yes

	hidden = yes
	
	trigger = {	#조건
	}
	immediate = {	
		##정통성
		if = {
			limit = {
				government_legitimacy < 25
			}
			activate_law = law_type:Illegitimate_Government
		}	
		
		if = {
			limit = {
				government_legitimacy >= 25
				government_legitimacy < 50
			}
			activate_law = law_type:Unacceptable_Government
		}	
		
		if = {
			limit = {
				government_legitimacy >= 50
				government_legitimacy < 75
			}
			activate_law = law_type:Contested_Government
		}		
		
		if = {
			limit = {
				government_legitimacy >= 75
				government_legitimacy < 90
			}
			activate_law = law_type:Legitimate_Government
		}				
		
		if = {
			limit = {
				government_legitimacy >= 90
				government_legitimacy <= 100
			}
			activate_law = law_type:Righteous_Government
		}				
	}
}
