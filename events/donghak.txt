#북접
donghak_events.1 = {
	type = country_event
	
	event_image = {
		texture = "gfx/event_pictures/seeking.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = donghak_events.1.t
	desc = donghak_events.1.d
	flavor = donghak_events.1.f

	duration = 3

	trigger = {
		exists = c:KOR
		c:KOR = ROOT
		NOT = { has_law = law_type:law_serfdom_banned }
		NOT = { has_variable = north_jeop }
         }
		
    immediate = {
		set_variable = {
			name = north_jeop
			value = yes
		}
	}
                                                  
	option = { #북접 모임
		name = donghak_events.1.a
        default_option = yes
        ig:ig_rural_folk = {
	        add_modifier = {
	            name = donghak_political_power
	            months = 12
	        }
		}
           }
	
}

	duration = 3
                
     trigger = {
		exists = c:KOR

	  NOT = { has_modifier = donghak_boeun_assembly_bad }
     }

	 immediate = {

	            name = ig_mass_arrests
	            months = 12
	        }
		}
                     
     }

        name = donghak_events.6.b
		 add_modifier = {
	            name = donghak_sangso_acceptance
	            months = 36
	        }
      }
 }

#고부민란
donghak_events.7 = {
	type = country_event

	event_image = {
		texture = "gfx/event_pictures/bonggi.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/slaves_night"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = donghak_events.7.t
	desc = donghak_events.7.d
	flavor = donghak_events.7.f

	duration = 3
                
        trigger = {
		has_insurrectionary_interest_groups = yes
		ig:ig_rural_folk = {
			is_revolutionary = no
			is_in_government = no
			is_marginal = no
    		AND = {
	    		ig_approval <= -1
	    		ig_approval > -8
	    	}
			NOT = { has_modifier = donghak_sangso_acceptance }
			NOT = { has_modifier = ig_mass_arrests }
		}
	}
    

	 immediate = {
		set_variable = {
			name =  gobu_rebel
			value = yes
			}
			ig:ig_rural_folk = {
			save_scope_as = ig_rf_on_the_edge
			}
			ig:ig_landowners = {
			limit = {
    			is_in_government = yes
			}
			leader = {
				save_scope_as = landowners_leader
		}
		}
		}

      option = { #진압한다
         default_option = yes
		name = donghak_events.7.a
		ig:ig_rural_folk = {
			
		 join_revolution = yes
		add_modifier = {
	            name = donghak_revolts
	            months = 20
	        }
			}

                     
     }

      option = {#요구를 받아들인다
          name = donghak_events.7.b
		  scope:landowners_leader = {
				kill_character = yes
			}
			ig:ig_landowners = {
	        add_modifier = {
	            name = donghak_revolts_landowners
	            months = 12
	        }
		}
		}

}
