jku.3 = { #일본 황실의 반응

    	type = country_event

	event_image = {
		texture = "gfx/event_pictures/sendball.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/slaves_night"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = jku.3.t
	desc = jku.3.d
	flavor = jku.3.f

   trigger = {
	has_global_variable = jku_royal_marriage
	exists = c:JAP
	this = c:JAP 
   }

   option = {#허락
      default_option = yes
      name = jku.3.a
	  set_global_variable = jku_acceptance
	  c:KOR = {
            trigger_event = { id = jku.5 days = 3 popup = yes }
        }
      if = {
			limit = { exists = c:KOR }
			change_relations = {
				country = c:KOR
				value = 50
			}
		}
   }

  option = {#거부
      name = jku.3.b
	  c:KOR = {
            trigger_event = { id = jku.4 days = 3 popup = yes }
        }
     if = {
		set_global_variable = jku_reject
			limit = { exists = c:KOR }
			change_relations = {
				country = c:KOR
				value = -50
			}
		}
   }
}
