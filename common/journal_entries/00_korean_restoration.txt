			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					ig:ig_landowners = {
						is_powerful = no

					}
					has_insurrectionary_interest_groups = no
					is_at_war = no
				}
				change_variable = {
					name = restart_var
					add = 1
				}
			}
		}
	}

	complete = {
		trigger_if = {
			limit = { exists = var:restart_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		ig:ig_landowners = {
			is_powerful = no
			is_in_government = no
		}
		has_insurrectionary_interest_groups = no
		is_at_war = no
	}

	on_complete = {
                                remove_modifier = hwangok_problem
		remove_modifier = jeonjeong_problem
		remove_modifier = gunjeong_problem
		trigger_event = {
			id = korean_restoration.2
		}
	}

	current_value = {
		value = root.var:restart_var
	}

	goal_add_value = {
		value = 120
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
