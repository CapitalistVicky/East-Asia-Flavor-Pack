# Bastion of the Landowners
# Can also appeal to the Devout, PB, and Industrialists

conservative_party = {
	color = { 108 64 179 }
	name = {
		first_valid = {
			triggered_desc = {
				desc = party_imperial_club
				trigger = {
					exists = c:KOR
					THIS = c:KOR
					has_law = law_type:law_monarchy
				}
			}
		}
	}
	join_weight = {
		if = { # capitalists more likely to join conservatives when the poor can vote
			limit = {
				has_ideology = ideology_moderate_scholar_paternalistic
				owner = {
					has_law = law_type:law_universal_suffrage
				}
			}
			add = {
				value = 5
				desc = "from_capitalists_suffrage"
			}
		}
		if = { # devout, industrialists more likely to join conservatives with public schools
			limit = {
				has_ideology = ideology_confucian
				owner = {
					has_law = law_type:law_public_schools
				}
			}
			add = {
				value = 2
				desc = "from_capitalists_schools"
			}
		}
   
		if = { # conservatives hate republics
			limit = {
				has_ideology = ideology_scholar_paternalistic
				owner = {
					NOR = {
						has_law = law_type:law_monarchy
						has_law = law_type:law_theocracy
					}
				}
			}
			add = {
				value = 3
				desc = "from_republic"
			}
		}
	}
}
